Quick Start

# Install web_all
pip install web_all

# Clone a website
web_all clone https://example.com --output my_site

# Analyze SEO
web_all seo analyze https://example.com

# Generate content
web_all content generate --topic "AI Technology" --type blog_post

# Start API server
web_all serve
